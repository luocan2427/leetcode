class Solution {
public:
    int numTrees(int n) {
        // Start typing your C/C++ solution below
        // DO NOT write int main() function
        vector<int> dp;
        dp.push_back(1);
        dp.push_back(1);
        for(int i = 2;i <= n;i++)
        {
            int num = 0;
            for(int j = 1;j <= i;j++)
            {
                num = num+dp[j-1]*dp[i-j];
            }
            dp.push_back(num);
        }
        return dp[n];
    }
};
