/**
 * Definition for binary tree
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int maxPathSum(TreeNode *root) {
        // Start typing your C/C++ solution below
        // DO NOT write int main() function
        if(root == NULL)
        {
            return 0;
        }
        int maxPath = root->val;
        maxSum(root, maxPath);
        return maxPath;
    }
    
    int maxSum(TreeNode *root, int &maxPath)
    {
        if(root == NULL)
        {
            return 0;
        }
        int left = maxSum(root->left,maxPath);
        int right = maxSum(root->right,maxPath);
        
        int res = root->val;
        int max = root->val;
        int temp;
        if(left > right)
        {
            temp = left;
        }
        else
        {
            temp = right;
        }
        if(temp > 0)
        {
            res += temp;
        }
        
        if(left > 0)
        {
            max += left;
        }
        if(right > 0)
        {
            max += right;
        }
        if(max > maxPath)
        {
            maxPath = max;
        }
        return res;
    }
};
