TreeNode *temp = NULL;
/**
 * Definition for binary tree
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    void flatten(TreeNode *root) {
        // Start typing your C/C++ solution below
        // DO NOT write int main() function
        if(root == NULL)
        {
            return;
        }
        temp = NULL;
        flatter(root);
    }
    void flatter(TreeNode *root)
    {
        if(root == NULL)
        {
            return;
        }
        flatter(root->right);
        flatter(root->left);
        root->left = NULL;
        root->right = temp;
        temp = root;
    }
};
